Arena Simulation Logic Snippet

This section shows a representative version of the Arena style discrete event logic used to model how patients move through intake, triage, treatment, and discharge. The AI generated forecasts were used as dynamic inputs to guide the simulation. 

// Representative Arena Simulation Flow Logic

CREATE Patient Arrival
  Arrival_Rate = AI_Arrival_Forecast[Hour, Day, Patient_Class]

ASSIGN Patient Attributes
  Patient_Class = AI_Class_Model(Historical_Arrivals, Acuity_Patterns)
  Acuity_Level = AI_Acuity_Model(Intake_Data)
  Expected_Length_Of_Stay = AI_LOS_Model(Patient_Class, Acuity_Level)

PROCESS Emergency Intake
  Resource = Intake_Staff
  Queue = Intake_Queue
  Record = Intake_Wait_Time

PROCESS Triage
  Resource = Triage_Nurse
  Queue = Triage_Queue
  Record = Triage_Wait_Time

DECIDE Treatment Path
  If Acuity_Level = High:
    Route_To_Critical_Treatment
  Else:
    Route_To_Standard_Treatment

PROCESS Treatment
  Resource = Provider + Treatment_Room
  Delay_Time = Expected_Length_Of_Stay
  Record = Provider_Utilization
  Record = Room_Utilization

PROCESS Discharge
  Resource = Discharge_Staff
  Queue = Discharge_Queue
  Record = Discharge_Delay

RECORD Final Metrics
  Average_Wait_Time
  Provider_Utilization
  Room_Utilization
  Daily_Throughput
  Service_Level_Target

AI Scenario Ranking
  Scenario_Score = Weighted_Model(
    Wait_Time,
    Provider_Utilization,
    Room_Utilization,
    Daily_Throughput,
    Bottleneck_Risk
  )

RECOMMEND Best_Staffing_And_Capacity_Scenario
